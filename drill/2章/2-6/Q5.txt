Q:ユーザテーブル（usr)、アンケート回答テーブル（quest）それぞれに含まれるユーザ氏名を比較し、
　双方に含まれるデータのみを取り出してみましょう。


A:
-- 選択命令文
 SELECT
-- 'ユーザ'テーブルの'氏'と'名'カラムを統合するため文字列結合して抽出する
  CONCAT(usrInfo.l_name, usrInfo.f_name) AS 氏名
-- 抽出元テーブルの指定
 FROM
-- 'ユーザ'テーブルを同名カラム名競合防止のため別名をつけて指定
  usr AS usrInfo
-- 抽出条件の指定
WHERE
-- 条件となるカラムの指定
 CONCAT(usrInfo.l_name, usrInfo.f_name)
-- サブクエリ結果に条件が含まれているデータを抽出するための演算子
IN
-- サブクエリの開始
 (
-- サブクエリの選択命令文
  SELECT
-- 抽出する'氏名'カラムの指定
   questInfo.name
-- 抽出元テーブルの指定
  FROM
-- 'アンケート回答テーブル'を同名カラム名競合防止のため別名をつけて指定
   quest AS questInfo
-- サブクエリの終了
  )
-- SQL命令の終了
;
